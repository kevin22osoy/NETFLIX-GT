CREATE TABLE tbl_METODO_DE_PAGO(
	ID_PAGO INTEGER NOT NULL,
	PLAN VARCHAR(15) NOT NULL,
	TIPO_DE_PLAN VARCHAR (45) NOT NULL,
	PRIMARY KEY(ID_PAGO),
);

CREATE TABLE tbl_CUENTA_NETFLIX(
	ID_USUARIO INTEGER NOT NULL,
	ID_PAGO INTEGER NOT NULL,8
	
	CORREO VARCHAR (35) NOT NULL,
	PRIMARY KEY(ID_USUARIO),
	FOREING KEY(ID_PAGO),
);

CREATE TABLE tbl_USUARIO(
	NOMBRE_DE_PERFIL VARCHAR(25) NOT NULL,
	ID_USUARIO INTEGER NOT NULL,
	PRIMARY KEY(NOMBRE_DE_PERFIL),
	FOREING KEY (ID_USUARIO),
);


CREATE TABLE tbl_ENC_PELICULA(
	NOMBRE_PELCULA INTEGER NOT NULL,
	INFORMACION_GENERAL VARCHAR (50) NOT NULL,
	NOMBRE_DE_PERFIL VARCHAR (25) NOT NULL,
	PRIMARY KEY(NOMBRE_PELICULA),
	FOREING KEY (NOMBRE_DE_PERFIL),
);

CREATE TABLDE tbl_DETALLE_PELICULA(
	INFORMACION_GENERAL VARCHAR (50) NOT NULL,
	DURACION VARCHAR (30) NOT NULL,
	PUBLICACION INTEGER NOT NULL,
	DIRECTOR VARCHAR (75) NOT NULL,
	REPARTO VARCHAR (50),
	PRIMARY KEY(INFORMACION_GENERAL)
);
